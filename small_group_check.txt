check_non_solvable:=function(n)
  local i,list,small_groups;
  list:=[];
  small_groups:=AllSmallGroups(n);
  for i in small_groups do
    if IsSolvable(i)=false then
      Add(list,check(i));
    fi;
  od;
  return list;
end;;

check_all:=function(m,n)
  local i,list;
  list:=[];
  for i in [m..n] do
    Add(list,check_non_solvable(i));
  od;
  return list;
end;;
